A8255_CON EQU 0606H
A8255_A EQU 0600H
A8255_B EQU 0602H
A8255_C EQU 0604H

DATA SEGMENT
TABLE1:
	DB 06H
	DB 5BH
	DB 4FH
	DB 66H
	DB 6DH
	DB 7DH
DATA ENDS
CODE SEGMENT
	ASSUME CS:CODE,DS:DATA

START:
	MOV AL,10000001B
	MOV DX,A8255_CON
	OUT DX,AL
	MOV AX,DATA
	MOV DS,AX
	LEA SI,TABLE1
	
X1:
	MOV BX,05H
	MOV CX,06H
	MOV AL,11011111B
X2:
	MOV DX,A8255_A
	OUT DX,AL
	PUSH AX
	MOV AL,[BX+SI]
	MOV DX,A8255_B
	OUT DX,AL
	POP AX
	ROR AL,1
	DEC BX
	;显示并且处理下一个位置
	CALL DELAY
	LOOP X2
	JMP X1

DELAY:
	PUSH CX
	MOV CX,0FFH
D1:
	LOOP D1
	MOV CX,0FFH
D2:
	LOOP D2
	POP CX
	RET
CODE ENDS
	END START
